generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model Student {
  id        Int      @id @default(autoincrement())
  name      String
  class     String
  enrolled  Boolean  @default(true)
}

model Teacher {
  id        Int      @id @default(autoincrement())
  name      String
  subject   String
}

model User {
  id        Int      @id @default(autoincrement())
  username  String   @unique
  password  String
  role      String
}
